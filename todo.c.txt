#include <stdio.h>
#include <string.h>

int main() {
    char tasks[50][100];
    int completed[50];
    int count = 0, choice, i, index;

    for(i = 0; i < 50; i++)
        completed[i] = 0;

    while(1) {
        printf("\n--- TO-DO LIST MENU ---\n");
        printf("1. Add Task\n");
        printf("2. View Tasks\n");
        printf("3. Mark Completed\n");
        printf("4. Delete Task\n");
        printf("5. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);
        getchar(); // to consume newline

        switch(choice) {
            case 1:
                if(count < 50) {
                    printf("Enter task: ");
                    fgets(tasks[count], 100, stdin);
                    tasks[count][strcspn(tasks[count], "\n")] = 0;
                    completed[count] = 0;
                    count++;
                } else {
                    printf("Task list is full!\n");
                }
                break;

            case 2:
                printf("\n--- Your Tasks ---\n");
                for(i = 0; i < count; i++) {
                    printf("%d. %s [%s]\n", i+1, tasks[i], completed[i] ? "Completed" : "Pending");
                }
                break;

            case 3:
                printf("Enter task number to mark completed: ");
                scanf("%d", &index);
                if(index >= 1 && index <= count) {
                    completed[index-1] = 1;
                } else {
                    printf("Invalid task number!\n");
                }
                break;

            case 4:
                printf("Enter task number to delete: ");
                scanf("%d", &index);
                if(index >= 1 && index <= count) {
                    for(i = index - 1; i < count - 1; i++) {
                        strcpy(tasks[i], tasks[i+1]);
                        completed[i] = completed[i+1];
                    }
                    count--;
                } else {
                    printf("Invalid task number!\n");
                }
                break;

            case 5:
                return 0;

            default:
                printf("Invalid choice!\n");
        }
    }
}
